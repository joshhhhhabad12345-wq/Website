local function uid()
    local rng   = Random.new()
    local chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789"
    local out   = ""
    for _ = 1, 16 do
        local r = rng:NextInteger(1, #chars)
        out = out .. chars:sub(r, r)
    end
    return out
end

local function new(class, props)
    local inst = Instance.new(class)
    if props then
        for k, v in pairs(props) do
            if k ~= "Parent" then pcall(function() inst[k] = v end) end
        end
        if props.Parent then inst.Parent = props.Parent end
    end
    return inst
end

local function addCorner(parent, radius)

local function addPadding(parent, t, b, l, r)

        PaddingTop    = UDim.new(0, t or 6),
        PaddingBottom = UDim.new(0, b or 6),
        PaddingLeft   = UDim.new(0, l or 10),
        PaddingRight  = UDim.new(0, r or 10),
        Parent = parent,
    })
end

local function safeHttp(url, method, body, headers)
    local req = (syn and syn.request) or (http and http.request) or request or http_request
    if req then
        local ok, res = pcall(req, {
            Url     = url,
            Method  = method or "GET",
            Body    = body,
            Headers = headers or {},
        })
        if ok and res then return res.Body, res.StatusCode end
        return nil, 0
    else
        local ok, res = pcall(function() return game:HttpGet(url) end)
        if ok then return res, 200 end
        return nil, 0
    end
end

local Scanner = {}
Scanner.__index = Scanner

function Scanner.new() return setmetatable({}, Scanner) end

function Scanner:FullPath(inst)
    local parts, current = {}, inst
    while current and current ~= game do
        local name = tostring(current.Name)
        table.insert(parts, 1, name:match("^[%a_][%w_]*$") and ("." .. name) or ('["' .. name .. '"]'))
        current = current.Parent
    end
    if #parts == 0 then return "game" end
    return "game" .. table.concat(parts)
end

function Scanner:ScanAll()
    local result = { scripts={}, remotes={}, modules={}, parts={}, players={} }
    local function walk(inst, depth)
        if depth > 12 then return end
        local ok, children = pcall(function() return inst:GetChildren() end)
        if not ok or not children then return end
        for _, child in ipairs(children) do
            local cls  = child.ClassName
            local path = self:FullPath(child)
            if cls == "LocalScript" or cls == "Script" then
                table.insert(result.scripts, { name=child.Name, class=cls, path=path, instance=child })
            elseif cls == "ModuleScript" then
                table.insert(result.modules, { name=child.Name, class=cls, path=path, instance=child })
            elseif cls == "RemoteEvent" or cls == "RemoteFunction" or cls == "BindableEvent" or cls == "BindableFunction" then
                table.insert(result.remotes, { name=child.Name, class=cls, path=path })
            elseif cls == "Part" or cls == "MeshPart" or cls == "UnionOperation" then
                table.insert(result.parts, { name=child.Name, class=cls, path=path })
            end
            walk(child, depth + 1)
        end
    end
    for _, svcName in ipairs(SCAN_TARGETS) do
        local ok, svc = pcall(game.GetService, game, svcName)
        if ok and svc then walk(svc, 0) end
    end
    for _, inst in ipairs(getnil_fn()) do
        local cls = inst.ClassName
        if cls == "LocalScript" or cls == "Script" then
            table.insert(result.scripts, { name=tostring(inst.Name), class=cls.." (nil)", path="[Nil]."..tostring(inst.Name), instance=inst })
        end
    end
    for _, plr in ipairs(Players:GetPlayers()) do
        table.insert(result.players, tostring(plr.Name))
    end
    return result
end

function Scanner:FormatReport(data)
    local lines = {}
    table.insert(lines, "=== GAME SCAN REPORT ===")
    table.insert(lines, ("Players (%d): %s"):format(#data.players, table.concat(data.players, ", ")))
    table.insert(lines, ("Scripts: %d | Remotes: %d | Modules: %d | Parts: %d"):format(
        #data.scripts, #data.remotes, #data.modules, #data.parts))
    table.insert(lines, "")
    if #data.scripts > 0 then
        table.insert(lines, "--- Scripts ---")
        for i, s in ipairs(data.scripts) do
            if i > 30 then table.insert(lines, ("  ...and %d more"):format(#data.scripts-30)); break end
            table.insert(lines, ("  [%s] %s"):format(s.class, s.path))
        end
        table.insert(lines, "")
    end
    if #data.remotes > 0 then
        table.insert(lines, "--- Remotes / Bindables ---")
        for i, r in ipairs(data.remotes) do
            if i > 30 then table.insert(lines, ("  ...and %d more"):format(#data.remotes-30)); break end
            table.insert(lines, ("  [%s] %s"):format(r.class, r.path))
        end
        table.insert(lines, "")
    end
    if #data.modules > 0 then
        table.insert(lines, "--- ModuleScripts ---")
        for i, m in ipairs(data.modules) do
            if i > 20 then table.insert(lines, ("  ...and %d more"):format(#data.modules-20)); break end
            table.insert(lines, ("  %s"):format(m.path))
        end
    end
    return table.concat(lines, "\n")
end

local Highlighter = {}
Highlighter.__index = Highlighter
function Highlighter.new() return setmetatable({}, Highlighter) end

local KW   = {["and"]=1,["break"]=1,["do"]=1,["else"]=1,["elseif"]=1,["end"]=1,
              ["for"]=1,["function"]=1,["if"]=1,["in"]=1,["local"]=1,["not"]=1,
              ["or"]=1,["repeat"]=1,["return"]=1,["then"]=1,["until"]=1,["while"]=1,["continue"]=1}
local LIT  = {["true"]=1,["false"]=1,["nil"]=1,["self"]=1}
local GLOB = {["game"]=1,["workspace"]=1,["math"]=1,["string"]=1,["table"]=1,["task"]=1,
              ["Enum"]=1,["Instance"]=1,["Color3"]=1,["Vector3"]=1,["CFrame"]=1,["UDim2"]=1,
              ["pcall"]=1,["pairs"]=1,["ipairs"]=1,["tostring"]=1,["tonumber"]=1,["print"]=1,
              ["warn"]=1,["error"]=1,["require"]=1,["setmetatable"]=1,["getmetatable"]=1,["type"]=1}

function Highlighter:Run(src)
    local function esc(s) return s:gsub("&","&amp;"):gsub("<","&lt;"):gsub(">","&gt;") end
    local out, i, len = {}, 1, #src
    while i <= len do
        local c = src:sub(i,i)
        if src:sub(i,i+1) == "--" then
            local j = src:find("\n", i, true)
            local sl = j and src:sub(i,j-1) or src:sub(i)
            table.insert(out, '<font color="#5C6370">'..esc(sl).."</font>")
            i = j and j or len+1
        elseif c == '"' or c == "'" then
            local q, j = c, i+1
            while j <= len do
                local b = src:sub(j,j)
                if b == "\\" then j=j+2
                elseif b == q then j=j+1; break
                elseif b == "\n" then break
                else j=j+1 end
            end
            table.insert(out, '<font color="#98C379">'..esc(src:sub(i,j-1)).."</font>")
            i = j
        elseif c:match("%d") then
            local j = i+1
            while j <= len and src:sub(j,j):match("[%d%.eExX_]") do j=j+1 end
            table.insert(out, '<font color="#D19A66">'..esc(src:sub(i,j-1)).."</font>")
            i = j
        elseif c:match("[%a_]") then
            local j = i+1
            while j <= len and src:sub(j,j):match("[%w_]") do j=j+1 end
            local word = src:sub(i,j-1)
            if KW[word] then
                table.insert(out, '<font color="#C678DD">'..word.."</font>")
            elseif LIT[word] then
                table.insert(out, '<font color="#D19A66">'..word.."</font>")
            elseif GLOB[word] then
                table.insert(out, '<font color="#61AFEF">'..word.."</font>")
            else
                table.insert(out, esc(word))
            end
            i = j
        elseif c == "\n" then
            table.insert(out, "\n"); i=i+1
        else
            table.insert(out, esc(c)); i=i+1
        end
    end
    return table.concat(out)
end

local function parseMarkdown(text)
    local segs, remaining = {}, text
    while #remaining > 0 do
        local s, e, lang, code = remaining:find("```([%w]*)\n?(.-)\n?```")
        if s then
            if s > 1 then table.insert(segs, { type="text", content=remaining:sub(1,s-1) }) end
            table.insert(segs, { type="code", lang=lang or "", content=code or "" })
            remaining = remaining:sub(e+1)
        else
            table.insert(segs, { type="text", content=remaining }); break
        end
    end
    return segs
end

local PuterClient = {}
PuterClient.__index = PuterClient

function PuterClient.new(modelName)
    return setmetatable({
        model     = modelName or DEFAULT_MODEL.name,
        history   = {},
        anonToken = nil,
    }, PuterClient)
end

function PuterClient:SetModel(modelName)
    self.model = modelName
end

function PuterClient:AddContext(role, text)
    -- OpenAI-style: role = "user" | "assistant" | "system"
    table.insert(self.history, { role = role, content = text })
end

function PuterClient:ClearHistory()
    self.history = {}
end

-- Get / reuse an anonymous Puter session token
function PuterClient:_EnsureToken()
    if self.anonToken then return self.anonToken end

    local body, status = safeHttp(
        PUTER_SIGNUP_URL, "POST",
        HttpService:JSONEncode({ is_temp = true }),
        { ["Content-Type"] = "application/json" }
    )

    if body and #body > 0 then
        local ok, data = pcall(function() return HttpService:JSONDecode(body) end)
        if ok and data then
            -- response may have token at top level or nested
            local tok = data.token or (data.body and data.body.token)
            if tok then
                self.anonToken = tostring(tok)
                return self.anonToken
            end
        end
    end

    -- Some executors block signup; attempt without token anyway
    return nil
end

function PuterClient:Send(userMessage)
    self:AddContext("user", userMessage)

    local token = self:_EnsureToken()

    local headers = { ["Content-Type"] = "application/json" }
    if token then
        headers["Authorization"] = "Bearer " .. token
    end

    local payload = HttpService:JSONEncode({
        interface = "puter-chat-completion",
        test_mode = false,
        method    = "complete",
        args      = {
            model    = self.model,
            messages = self.history,
        },
    })

    local body, status = safeHttp(PUTER_AI_URL, "POST", payload, headers)

    if not body or #body == 0 then
        return nil, ("HTTP error – status %d. Check executor HTTP permissions."):format(status)
    end

    local ok, decoded = pcall(function() return HttpService:JSONDecode(body) end)
    if not ok or not decoded then
        return nil, "JSON decode failed – raw: " .. tostring(body):sub(1, 200)
    end

    -- Puter wraps the response in { result: { ... } }
    local inner = decoded.result or decoded

    -- Check for API-level error
    if decoded.error then
        local msg = decoded.error
        if type(msg) == "table" then msg = msg.message or msg.code or "API error" end
        return nil, tostring(msg)
    end

    -- Extract text from OpenAI-compatible shape
    local text = ""
    if inner.choices and inner.choices[1] then
        local choice = inner.choices[1]
        local msg = choice.message or choice.delta
        if msg then
            text = tostring(msg.content or "")
        end
    elseif inner.content then
        -- Anthropic shape
        if type(inner.content) == "table" then
            for _, part in ipairs(inner.content) do
                if part.text then text = text .. part.text end
            end
        else
            text = tostring(inner.content)
        end
    elseif inner.text then
        text = tostring(inner.text)
    end

    if #text == 0 then
        -- Last resort: dump raw for debugging
        return nil, "Empty response. Raw: " .. HttpService:JSONEncode(inner):sub(1, 300)
    end

    self:AddContext("assistant", text)
    return text, nil
end

-- ─────────────────────────────
-- 10c. GAME CONTEXT
-- ─────────────────────────────
function PuterAIChat:_BuildGameContext()
    local data   = self._scanner:ScanAll()
    local report = self._scanner:FormatReport(data)
    local gameName = "Unknown"
    pcall(function() gameName = game:GetService("MarketplaceService"):GetProductInfo(game.PlaceId).Name end)
    if not gameName or gameName == "" then pcall(function() gameName = game.Name end) end
    return ("You are an AI assistant integrated into a Roblox game session.\n"
        .."Game: "..tostring(gameName).." | PlaceId: "..tostring(game.PlaceId)
        .." | JobId: "..tostring(game.JobId).."\n\n"
        ..report.."\n\n"
        .."Help the user understand, analyze, and work with this Roblox game. "
        .."You have deep knowledge of Luau, Roblox APIs, game design, and exploit development. "
        .."When showing code, always use ``` code blocks.")
end

        0, math.max(0, self.ChatScrollFrame.AbsoluteCanvasSize.Y - self.ChatScrollFrame.AbsoluteSize.Y))

        pcall(function() self._thinkingFrame:Destroy() end)
        self._thinkingFrame = nil
        self._thinkingLabel = nil
    end
end

-- ─────────────────────────────
-- 10f. SEND HANDLER + COMMANDS
-- ─────────────────────────────
function PuterAIChat:_OnSend()
    if self._thinking then return end
    if not self._client then
        self:_AddMessage("ai", "Please select a model first using the setup screen.")
        return
    end

    local text = self.InputBox.Text:gsub("^%s+",""):gsub("%s+$","")
    if #text == 0 then return end
    self.InputBox.Text = ""
    self:_AddMessage("user", text)

    -- Built-in commands
    if text:lower() == "/scan" then self:_ScanGame(); return end

    if text:lower():sub(1,11) == "/decompile " then
        self:_DecompileByPath(text:sub(12)); return
    end

    if text:lower() == "/clear" then
        self._client:ClearHistory()
        self._client:AddContext("system", self:_BuildGameContext())
        self:_AddMessage("ai", "Conversation cleared. Game context reloaded.")
        return
    end

    if text:lower() == "/help" then
        self:_AddMessage("ai",
            "**Commands:**\n"..
            "```\n"..
            "/scan              — Scan game for scripts, remotes, modules\n"..
            "/decompile <path>  — Decompile a script by game path\n"..
            "                     e.g. /decompile game.Workspace.MyScript\n"..
            "/clear             — Clear conversation history\n"..
            "/help              — Show this message\n"..
            "```\n"..
            "Or just chat normally — ask anything about Roblox, Luau, or the game!"
        )
        return
    end

    -- Send to Puter AI
    self._thinking = true
    self.SendBtn.BackgroundColor3 = T.ScanBtn
    self:_ShowThinking()

    task.spawn(function()
        local reply, err = self._client:Send(text)
        task.wait()
        self:_HideThinking()
        self._thinking = false
        self.SendBtn.BackgroundColor3 = T.SendBtn

        if err then
            self:_AddMessage("ai",
                "Error: " .. tostring(err) ..
                "\n\nTips:\n- Make sure your executor allows HTTPS requests\n"..
                "- Try `/clear` to reset conversation\n- Try switching models")
        else
            self:_AddMessage("ai", reply or "(empty response)")
        end
    end)
end

-- ─────────────────────────────
-- 10h. DECOMPILE
-- ─────────────────────────────
function PuterAIChat:_DecompileByPath(pathStr)
    if self._thinking then return end
    self._thinking = true
    self:_ShowThinking()

    task.spawn(function()
        local target = nil
        pcall(function()
            target = game
            for part in pathStr:gmatch("[^%.%[%]\"]+") do
                if part ~= "game" and part ~= "" then
                    local ok, child = pcall(function() return target:FindFirstChild(part) or target[part] end)
                    if ok and child then target = child else break end
                end
            end
        end)

        self:_HideThinking()
        self._thinking = false

        if not target or target == game then
            self:_AddMessage("ai", "Could not find: `"..pathStr.."`\n\nTip: run `/scan` first to see valid paths.")
            return
        end

        local cls = tostring(target.ClassName)
        if cls ~= "LocalScript" and cls ~= "Script" and cls ~= "ModuleScript" then
            self:_AddMessage("ai", "`"..pathStr.."` is a `"..cls.."` — only scripts can be decompiled.")
            return
        end

        self:_AddMessage("ai", "Decompiling `"..tostring(target.Name).."` (`"..cls.."`)...")
        self._thinking = true
        self:_ShowThinking()

        task.spawn(function()
            local source, ok = "", false
            pcall(function() source = target.Source; if #source > 0 then ok = true end end)
            if not ok then
                pcall(function() source = decompile_fn(target); if #source > 0 then ok = true end end)
            end

            task.wait()
            self:_HideThinking()
            self._thinking = false

            if not ok or #source == 0 then
                self:_AddMessage("ai", "Could not decompile `"..tostring(target.Name).."`.\n\n"..
                    "Possible reasons:\n- Script is obfuscated\n- Executor doesn't support decompile()\n- Script is protected")
                return
            end

            local display = #source > 8000 and (source:sub(1,8000).."\n-- [truncated, "..#source.." chars total]") or source
            self:_AddMessage("ai", "**Decompiled: `"..tostring(target.Name).."` ("..cls..")**\n```lua\n"..display.."\n```")

            if self._client then
                local snippet = #source > 3000 and source:sub(1,3000).."\n-- [truncated]" or source
                self._client:AddContext("user", "I decompiled `"..pathStr.."`. Source:\n```lua\n"..snippet.."\n```\nAnalyze it.")
                self._client:AddContext("assistant", "Source loaded. Ask me anything about it.")
                self:_AddMessage("ai", "Source loaded into context — ask me to analyze, explain, or modify it!")
            end
        end)
    end)
end
