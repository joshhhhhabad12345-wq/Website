local function uid()
    local rng = Random.new()
    local chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789"
    local out = ""
    for _ = 1, 16 do
        local r = rng:NextInteger(1, #chars)
        out = out .. chars:sub(r, r)
    end
    return out
end

local function new(class, props)
    local inst = Instance.new(class)
    if props then
        for k, v in pairs(props) do
            if k ~= "Parent" then
                pcall(function() inst[k] = v end)
            end
        end
        if props.Parent then
            inst.Parent = props.Parent
        end
    end
    return inst
end

local function addCorner(parent, radius)
    new("UICorner", { CornerRadius = UDim.new(0, radius or 8), Parent = parent })
end

local function addPadding(parent, t, b, l, r)
    new("UIPadding", {
        PaddingTop = UDim.new(0, t or 6),
        PaddingBottom = UDim.new(0, b or 6),
        PaddingLeft = UDim.new(0, l or 10),
        PaddingRight = UDim.new(0, r or 10),
        Parent = parent,
    })
end

local function safeHttp(url, method, body, headers)
    local req = (syn and syn.request) or (http and http.request) or request or http_request
    if req then
        local ok, res = pcall(req, {
            Url = url,
            Method = method or "GET",
            Body = body,
            Headers = headers or {},
        })
        if ok and res then
            return res.Body, res.StatusCode
        end
        return nil, 0
    else
        local ok, res = pcall(function()
            return game:HttpGet(url)
        end)
        if ok then
            return res, 200
        end
        return nil, 0
    end
end

local Scanner = {}
Scanner.__index = Scanner

function Scanner.new()
    return setmetatable({}, Scanner)
end

function Scanner:ScanAll()
    local result = {
        scripts = {},
        remotes = {},
        modules = {},
        parts   = {},
        players = {},
        summary = {},
    }

    local function walk(inst, depth)
        if depth > 12 then return end
        local ok, children = pcall(function() return inst:GetChildren() end)
        if not ok or not children then return end
        for _, child in ipairs(children) do
            local cls = child.ClassName
            local name = tostring(child.Name)

            if cls == "LocalScript" or cls == "Script" or cls == "ModuleScript" then
                local entry = {
                    name = name,
                    class = cls,
                    path = self:FullPath(child),
                    instance = child,
                }
                if cls == "ModuleScript" then
                    table.insert(result.modules, entry)
                else
                    table.insert(result.scripts, entry)
                end
            elseif cls == "RemoteEvent" or cls == "RemoteFunction" or cls == "BindableEvent" or cls == "BindableFunction" then
                table.insert(result.remotes, {
                    name = name,
                    class = cls,
                    path = self:FullPath(child),
                    instance = child,
                })
            elseif cls == "Part" or cls == "MeshPart" or cls == "UnionOperation" or cls == "SpecialMesh" then
                table.insert(result.parts, {
                    name = name,
                    class = cls,
                    path = self:FullPath(child),
                })
            end

            walk(child, depth + 1)
        end
    end

    for _, svcName in ipairs(SCAN_TARGETS) do
        local ok, svc = pcall(game.GetService, game, svcName)
        if ok and svc then
            walk(svc, 0)
        end
    end

    local nilInsts = getnil_fn()
    for _, inst in ipairs(nilInsts) do
        local cls = inst.ClassName
        if cls == "LocalScript" or cls == "Script" then
            table.insert(result.scripts, {
                name = tostring(inst.Name),
                class = cls .. " (nil)",
                path = "[Nil]." .. tostring(inst.Name),
                instance = inst,
            })
        end
    end

    for _, plr in ipairs(Players:GetPlayers()) do
        table.insert(result.players, tostring(plr.Name))
    end

    result.summary = {
        scripts = #result.scripts,
        remotes = #result.remotes,
        modules = #result.modules,
        parts   = #result.parts,
        players = #result.players,
    }

    return result
end

function Scanner:FullPath(inst)
    local parts, current = {}, inst
    while current and current ~= game do
        local name = tostring(current.Name)
        if name:match("^[%a_][%w_]*$") then
            table.insert(parts, 1, "." .. name)
        else
            table.insert(parts, 1, '["' .. name .. '"]')
        end
        current = current.Parent
    end
    if #parts == 0 then return "game" end
    return "game" .. table.concat(parts)
end

function Scanner:FormatReport(data)
    local lines = {}
    table.insert(lines, "=== GAME SCAN REPORT ===")
    table.insert(lines, ("Players online: %d → %s"):format(
        data.summary.players,
        table.concat(data.players, ", ")
    ))
    table.insert(lines, ("Scripts: %d | Remotes: %d | Modules: %d | Parts: %d"):format(
        data.summary.scripts,
        data.summary.remotes,
        data.summary.modules,
        data.summary.parts
    ))
    table.insert(lines, "")

    if #data.scripts > 0 then
        table.insert(lines, "--- Scripts ---")
        for i, s in ipairs(data.scripts) do
            if i > 30 then
                table.insert(lines, ("  ... and %d more"):format(#data.scripts - 30))
                break
            end
            table.insert(lines, ("  [%s] %s"):format(s.class, s.path))
        end
        table.insert(lines, "")
    end

    if #data.remotes > 0 then
        table.insert(lines, "--- Remotes / Bindables ---")
        for i, r in ipairs(data.remotes) do
            if i > 30 then
                table.insert(lines, ("  ... and %d more"):format(#data.remotes - 30))
                break
            end
            table.insert(lines, ("  [%s] %s"):format(r.class, r.path))
        end
        table.insert(lines, "")
    end

    if #data.modules > 0 then
        table.insert(lines, "--- ModuleScripts ---")
        for i, m in ipairs(data.modules) do
            if i > 20 then
                table.insert(lines, ("  ... and %d more"):format(#data.modules - 20))
                break
            end
            table.insert(lines, ("  %s"):format(m.path))
        end
    end

    return table.concat(lines, "\n")
end

local KW = {
    ["and"]=true,["break"]=true,["do"]=true,["else"]=true,["elseif"]=true,
    ["end"]=true,["for"]=true,["function"]=true,["if"]=true,["in"]=true,
    ["local"]=true,["not"]=true,["or"]=true,["repeat"]=true,["return"]=true,
    ["then"]=true,["until"]=true,["while"]=true,["continue"]=true,["export"]=true,
}
local LIT = {["true"]=true,["false"]=true,["nil"]=true,["self"]=true}
local GLOB = {
    ["game"]=true,["workspace"]=true,["script"]=true,["math"]=true,["string"]=true,
    ["table"]=true,["task"]=true,["Enum"]=true,["Instance"]=true,["Color3"]=true,
    ["Vector3"]=true,["Vector2"]=true,["CFrame"]=true,["UDim2"]=true,["UDim"]=true,
    ["pcall"]=true,["xpcall"]=true,["coroutine"]=true,["pairs"]=true,["ipairs"]=true,
    ["tostring"]=true,["tonumber"]=true,["print"]=true,["warn"]=true,["error"]=true,
    ["require"]=true,["setmetatable"]=true,["getmetatable"]=true,["type"]=true,
    ["typeof"]=true,["rawget"]=true,["rawset"]=true,["select"]=true,["assert"]=true,
    ["loadstring"]=true,["tick"]=true,["time"]=true,["os"]=true,["spawn"]=true,
}

function Highlighter:Run(src)
    local function esc(s)
        return s:gsub("&","&amp;"):gsub("<","&lt;"):gsub(">","&gt;")
    end
    local out = {}
    local i = 1
    local len = #src

    while i <= len do
        local c = src:sub(i, i)

        -- line comment
        if src:sub(i, i+1) == "--" then
            local j = src:find("\n", i, true)
            local slice = j and src:sub(i, j-1) or src:sub(i)
            table.insert(out, '<font color="#5C6370">' .. esc(slice) .. "</font>")
            i = j and j or len + 1

        -- string literal
        elseif c == '"' or c == "'" then
            local q = c
            local j = i + 1
            while j <= len do
                local b = src:sub(j,j)
                if b == "\\" then j = j + 2
                elseif b == q then j = j + 1; break
                elseif b == "\n" then break
                else j = j + 1 end
            end
            table.insert(out, '<font color="#98C379">' .. esc(src:sub(i, j-1)) .. "</font>")
            i = j

        -- number literal
        elseif c:match("%d") then
            local j = i + 1
            while j <= len and src:sub(j,j):match("[%d%.xXeE_%+%-]") do j = j + 1 end
            table.insert(out, '<font color="#D19A66">' .. esc(src:sub(i, j-1)) .. "</font>")
            i = j

        -- identifier / keyword
        elseif c:match("[%a_]") then
            local j = i + 1
            while j <= len and src:sub(j,j):match("[%w_]") do j = j + 1 end
            local word = src:sub(i, j-1)
            if KW[word] then
                table.insert(out, '<font color="#C678DD">' .. word .. "</font>")
            elseif LIT[word] then
                table.insert(out, '<font color="#D19A66">' .. word .. "</font>")
            elseif GLOB[word] then
                table.insert(out, '<font color="#61AFEF">' .. word .. "</font>")
            else
                table.insert(out, esc(word))
            end
            i = j

        -- newline
        elseif c == "\n" then
            table.insert(out, "\n")
            i = i + 1

        else
            table.insert(out, esc(c))
            i = i + 1
        end
    end

    return table.concat(out)
end

-- 8. MARKDOWN CODE BLOCK PARSER
--    Splits text into segments: {type="text"|"code", lang=string, content=string}
-- ===========================
local function parseMarkdown(text)
    local segments = {}
    local remaining = text
    while #remaining > 0 do
        local s, e, lang, code = remaining:find("```([%w]*)\n?(.-)\n?```")
        if s then
            if s > 1 then
                table.insert(segments, { type = "text", content = remaining:sub(1, s - 1) })
            end
            table.insert(segments, { type = "code", lang = lang or "", content = code or "" })
            remaining = remaining:sub(e + 1)
        else
            table.insert(segments, { type = "text", content = remaining })
            break
        end
    end
    return segments
end

local GeminiClient = {}
GeminiClient.__index = GeminiClient

function GeminiClient.new(apiKey, model)
    return setmetatable({
        apiKey  = apiKey or "",
        model   = model or "gemma-3-27b-it",
        history = {},
    }, GeminiClient)
end

function GeminiClient:SetModel(model)
    self.model = model
end

function GeminiClient:IsGemma()
    return not modelNeedsKey(self.model)
end

function GeminiClient:AddContext(role, text)
    table.insert(self.history, { role = role, parts = {{ text = text }} })
end

function GeminiClient:ClearHistory()
    self.history = {}
end

function GeminiClient:Send(userMessage)
    self:AddContext("user", userMessage)

    local payload = HttpService:JSONEncode({
        contents = self.history,
        generationConfig = {
            temperature = 0.85,
            maxOutputTokens = 8192,
        },
    })

    -- Gemma 3 models don't need an API key
    local url
    if self:IsGemma() then
        url = API_BASE .. self.model .. ":generateContent"
    else
        url = API_BASE .. self.model .. ":generateContent?key=" .. self.apiKey
    end

    local body, status = safeHttp(url, "POST", payload, {
        ["Content-Type"] = "application/json",
    })

    if not body then
        return nil, "HTTP request failed (status " .. tostring(status) .. ")"
    end

    local ok, decoded = pcall(function()
        return HttpService:JSONDecode(body)
    end)

    if not ok or not decoded then
        return nil, "Failed to decode response"
    end

    if decoded.error then
        return nil, tostring(decoded.error.message or "API error")
    end

    local text = ""
    if decoded.candidates and decoded.candidates[1] then
        local cand = decoded.candidates[1]
        if cand.content and cand.content.parts then
            for _, part in ipairs(cand.content.parts) do
                if part.text then
                    text = text .. part.text
                end
            end
        end
    end

    if #text == 0 then
        return nil, "Empty response from Gemini"
    end

    self:AddContext("model", text)
    return text, nil
end

-- ─────────────────────────────
-- 10c. GAME CONTEXT BUILDER
-- ─────────────────────────────
function GeminiAIChat:_BuildGameContext()
    local data = self._scanner:ScanAll()
    local report = self._scanner:FormatReport(data)
    local gameName = "Unknown"
    pcall(function() gameName = game:GetService("MarketplaceService"):GetProductInfo(game.PlaceId).Name end)
    if not gameName or gameName == "" then
        pcall(function() gameName = game.Name end)
    end

    return ("You are an AI assistant integrated into a Roblox game session.\n"
        .. "Game Name: " .. tostring(gameName) .. "\n"
        .. "Place ID: " .. tostring(game.PlaceId) .. "\n"
        .. "Job ID: " .. tostring(game.JobId) .. "\n\n"
        .. "Current game scan:\n" .. report .. "\n\n"
        .. "Help the user understand and analyze this Roblox game. "
        .. "You have knowledge of Luau, Roblox APIs, game design, and exploit development. "
        .. "When showing code, always use ``` code blocks. "
        .. "Be concise but thorough.")
end

-- ─────────────────────────────
-- 10d. CHAT MESSAGE RENDERER
-- ─────────────────────────────
function GeminiAIChat:_AddMessage(role, text)
    local isUser = role == "user"
    local layoutOrder = #self._messages + 1
    table.insert(self._messages, { role = role, text = text })

    -- Outer wrapper (for alignment)
    local wrapper = new("Frame", {
        Name = "Msg_" .. layoutOrder,
        BackgroundTransparency = 1,
        Size = UDim2.new(1, 0, 0, 0),
        AutomaticSize = Enum.AutomaticSize.Y,
        LayoutOrder = layoutOrder,
        ZIndex = 2,
        Parent = self.ChatScrollFrame,
    })

    -- Avatar label
    local avatarLbl = new("TextLabel", {
        Text = isUser and "You" or "AI",
        Font = Enum.Font.GothamBold,
        TextSize = 10,
        TextColor3 = isUser and T.UserBubble or T.Highlight,
        BackgroundTransparency = 1,
        Size = UDim2.new(0, 30, 0, 14),
        Position = isUser and UDim2.new(1, -40, 0, 0) or UDim2.new(0, 0, 0, 0),
        ZIndex = 3,
        Parent = wrapper,
    })
    _ = avatarLbl

    -- Bubble container
    local bubbleContainer = new("Frame", {
        BackgroundTransparency = 1,
        Size = UDim2.new(1, 0, 0, 0),
        AutomaticSize = Enum.AutomaticSize.Y,
        Position = UDim2.new(0, 0, 0, 16),
        ZIndex = 2,
        Parent = wrapper,
    })

    -- Parse markdown and build segments
    local segments = parseMarkdown(text)

    for _, seg in ipairs(segments) do
        if seg.type == "text" and #seg.content:gsub("%s", "") > 0 then
            -- Regular text bubble
            local bubble = new("TextLabel", {
                Text = seg.content,
                Font = Enum.Font.Gotham,
                TextSize = 13,
                TextColor3 = T.Text,
                BackgroundColor3 = isUser and T.UserBubble or T.AiBubble,
                Size = UDim2.new(0.85, 0, 0, 0),
                AutomaticSize = Enum.AutomaticSize.Y,
                AnchorPoint = isUser and Vector2.new(1, 0) or Vector2.new(0, 0),
                Position = isUser and UDim2.new(1, 0, 0, 0) or UDim2.new(0, 0, 0, 0),
                TextWrapped = true,
                RichText = false,
                TextXAlignment = Enum.TextXAlignment.Left,
                TextYAlignment = Enum.TextYAlignment.Top,
                BorderSizePixel = 0,
                ZIndex = 3,
                Parent = bubbleContainer,
            })
            addCorner(bubble, 10)
            addPadding(bubble, 8, 8, 12, 12)

            if isUser then
                new("UIStroke", { Color = Color3.fromRGB(0, 80, 160), Thickness = 1, ApplyStrokeMode = Enum.ApplyStrokeMode.Border, Parent = bubble })
            end

        elseif seg.type == "code" then
            -- Code block
            local codeWrap = new("Frame", {
                BackgroundColor3 = T.CodeBg,
                Size = UDim2.new(1, 0, 0, 0),
                AutomaticSize = Enum.AutomaticSize.Y,
                BorderSizePixel = 0,
                ZIndex = 3,
                Parent = bubbleContainer,
            })
            addCorner(codeWrap, 6)
            new("UIStroke", { Color = Color3.fromRGB(50, 60, 75), Thickness = 1, ApplyStrokeMode = Enum.ApplyStrokeMode.Border, Parent = codeWrap })

            -- Language label
            if seg.lang and #seg.lang > 0 then
                local langLbl = new("TextLabel", {
                    Text = seg.lang,
                    Font = Enum.Font.Gotham,
                    TextSize = 10,
                    TextColor3 = T.Highlight,
                    BackgroundColor3 = Color3.fromRGB(18, 22, 30),
                    Size = UDim2.new(1, 0, 0, 20),
                    BorderSizePixel = 0,
                    ZIndex = 4,
                    Parent = codeWrap,
                })
                addCorner(langLbl, 4)
            end

            -- Try syntax highlighting for Luau/lua
            local isLuau = seg.lang == "lua" or seg.lang == "luau" or seg.lang == ""
            local displayText = seg.content

            if isLuau and #displayText < 8000 then
                local hl = self._hl:Run(displayText)
                local codeLabel = new("TextLabel", {
                    Text = hl,
                    Font = Enum.Font.Code,
                    TextSize = 12,
                    TextColor3 = T.CodeText,
                    BackgroundTransparency = 1,
                    Size = UDim2.new(1, 0, 0, 0),
                    AutomaticSize = Enum.AutomaticSize.Y,
                    Position = UDim2.new(0, 0, 0, (seg.lang and #seg.lang > 0) and 20 or 0),
                    TextWrapped = true,
                    RichText = true,
                    TextXAlignment = Enum.TextXAlignment.Left,
                    TextYAlignment = Enum.TextYAlignment.Top,
                    BorderSizePixel = 0,
                    ZIndex = 4,
                    Parent = codeWrap,
                })
                addPadding(codeLabel, 6, 8, 10, 10)
            else
                local codeLabel = new("TextLabel", {
                    Text = displayText:sub(1, 5000),
                    Font = Enum.Font.Code,
                    TextSize = 12,
                    TextColor3 = T.CodeText,
                    BackgroundTransparency = 1,
                    Size = UDim2.new(1, 0, 0, 0),
                    AutomaticSize = Enum.AutomaticSize.Y,
                    Position = UDim2.new(0, 0, 0, (seg.lang and #seg.lang > 0) and 20 or 0),
                    TextWrapped = true,
                    RichText = false,
                    TextXAlignment = Enum.TextXAlignment.Left,
                    TextYAlignment = Enum.TextYAlignment.Top,
                    BorderSizePixel = 0,
                    ZIndex = 4,
                    Parent = codeWrap,
                })
                addPadding(codeLabel, 6, 8, 10, 10)
            end
        end
    end

    -- Auto-scroll to bottom
    task.wait()
    self.ChatScrollFrame.CanvasPosition = Vector2.new(
        0,
        math.max(0, self.ChatScrollFrame.AbsoluteCanvasSize.Y - self.ChatScrollFrame.AbsoluteSize.Y)
    )
end

-- ─────────────────────────────
-- 10e. THINKING INDICATOR
-- ─────────────────────────────
function GeminiAIChat:_ShowThinking()
    if self._thinkingFrame then return end
    local layoutOrder = #self._messages + 999
    local wrap = new("Frame", {
        Name = "ThinkingFrame",
        BackgroundTransparency = 1,
        Size = UDim2.new(1, 0, 0, 40),
        LayoutOrder = layoutOrder,
        ZIndex = 2,
        Parent = self.ChatScrollFrame,
    })
    self._thinkingFrame = wrap

    local dot = new("Frame", {
        BackgroundColor3 = T.AiBubble,
        Size = UDim2.new(0, 80, 0, 32),
        Position = UDim2.new(0, 0, 0, 4),
        BorderSizePixel = 0,
        ZIndex = 3,
        Parent = wrap,
    })
    addCorner(dot, 10)

    self._thinkingLabel = new("TextLabel", {
        Text = "● ● ●",
        Font = Enum.Font.GothamBold,
        TextSize = 14,
        TextColor3 = T.SubText,
        BackgroundTransparency = 1,
        Size = UDim2.new(1, 0, 1, 0),
        ZIndex = 4,
        Parent = dot,
    })

    task.spawn(function()
        local dots = {"●", "● ●", "● ● ●"}
        local i = 1
        while self._thinkingFrame and self._alive do
            if self._thinkingLabel then
                self._thinkingLabel.Text = dots[i]
            end
            i = (i % 3) + 1
            task.wait(0.4)
        end
    end)

    task.wait()
    self.ChatScrollFrame.CanvasPosition = Vector2.new(
        0,
        math.max(0, self.ChatScrollFrame.AbsoluteCanvasSize.Y - self.ChatScrollFrame.AbsoluteSize.Y)
    )
end

function GeminiAIChat:_HideThinking()
    if self._thinkingFrame then
        pcall(function() self._thinkingFrame:Destroy() end)
        self._thinkingFrame = nil
        self._thinkingLabel = nil
    end
end

-- ─────────────────────────────
-- 10f. SEND HANDLER + COMMANDS
-- ─────────────────────────────
function GeminiAIChat:_OnSend()
    if self._thinking then return end
    if not self._client then
        -- Auto-connect with Gemma 3 if no key is needed
        if not modelNeedsKey(self._selectedModel) then
            self._client = GeminiClient.new("", self._selectedModel)
            local gameCtx = self:_BuildGameContext()
            self._client:AddContext("user", gameCtx)
            self._client:AddContext("model", "Understood. I am an AI integrated with the Roblox game session.")
        else
            self:_AddMessage("ai", "Please enter your API key first, or switch to a Gemma 3 model (no key needed).")
            return
        end
    end

    local text = self.InputBox.Text:gsub("^%s+", ""):gsub("%s+$", "")
    if #text == 0 then return end

    self.InputBox.Text = ""
    self:_AddMessage("user", text)

    -- Built-in commands
    if text:lower() == "/scan" then
        self:_ScanGame()
        return
    end

    if text:lower():sub(1, 11) == "/decompile " then
        local path = text:sub(12)
        self:_DecompileByPath(path)
        return
    end

    if text:lower() == "/clear" then
        self._client:ClearHistory()
        local gameCtx = self:_BuildGameContext()
        self._client:AddContext("user", gameCtx)
        self._client:AddContext("model", "Understood. History cleared. I still have game context.")
        self:_AddMessage("ai", "Chat history cleared. I still have game context loaded.")
        return
    end

    if text:lower() == "/help" then
        self:_AddMessage("ai", "**Commands:**\n```\n/scan          - Scan game for scripts, remotes, modules\n/decompile <path> - Decompile a script by game path\n                  (e.g. /decompile game.Workspace.MyScript)\n/clear         - Clear conversation history\n/help          - Show this message\n```\n\nYou can also just chat normally! Ask about game mechanics, Luau code, or anything else.")
        return
    end

    -- Send to Gemini
    self._thinking = true
    self.SendBtn.BackgroundColor3 = T.ScanBtn
    self:_ShowThinking()

    task.spawn(function()
        local reply, err = self._client:Send(text)
        task.wait()
        self:_HideThinking()
        self._thinking = false
        self.SendBtn.BackgroundColor3 = T.SendBtn

        if err then
            self:_AddMessage("ai", "Error: " .. tostring(err) .. "\n\nMake sure your API key is valid and the selected model is available.")
        else
            self:_AddMessage("ai", reply or "(no response)")
        end
    end)
end

-- ─────────────────────────────
-- 10g. SCAN GAME COMMAND
-- ─────────────────────────────
function GeminiAIChat:_ScanGame()
    if self._thinking then return end
    self._thinking = true
    self.ScanBtn.BackgroundColor3 = T.ScanBtnHv
    self.ScanBtn.Text = "Scanning..."
    self:_ShowThinking()

    task.spawn(function()
        local data = self._scanner:ScanAll()
        local report = self._scanner:FormatReport(data)
        task.wait()
        self:_HideThinking()
        self._thinking = false
        self.ScanBtn.BackgroundColor3 = T.ScanBtn
        self.ScanBtn.Text = "⟳ Scan Game"

        self:_AddMessage("ai", "**Game Scan Complete!**\n\n```\n" .. report .. "\n```")

        if self._client then
            self._client:AddContext("user", "Here is an updated game scan:\n" .. report)
            self._client:AddContext("model", "Updated game scan received and stored in context.")
        end
    end)
end

-- ─────────────────────────────
-- 10h. DECOMPILE COMMAND
-- ─────────────────────────────
function GeminiAIChat:_DecompileByPath(pathStr)
    self._thinking = true
    self:_ShowThinking()

    task.spawn(function()
        local target = nil
        pcall(function()
            target = game
            for part in pathStr:gmatch("[^%.%[%]\"]+") do
                if part ~= "game" and part ~= "" then
                    local ok, child = pcall(function()
                        return target:FindFirstChild(part) or target[part]
                    end)
                    if ok and child then
                        target = child
                    else
                        break
                    end
                end
            end
        end)

        self:_HideThinking()
        self._thinking = false

        if not target or target == game then
            self:_AddMessage("ai", "Could not find instance at path: `" .. pathStr .. "`\n\nTip: Try `/scan` first to find valid paths.")
            return
        end

        local cls = tostring(target.ClassName)
        if cls ~= "LocalScript" and cls ~= "Script" and cls ~= "ModuleScript" then
            self:_AddMessage("ai", "Instance `" .. pathStr .. "` is a `" .. cls .. "`, not a Script.\n\nOnly LocalScript, Script, and ModuleScript can be decompiled.")
            return
        end

        self:_AddMessage("ai", "Decompiling `" .. tostring(target.Name) .. "` (`" .. cls .. "`)...")
        self:_ShowThinking()

        task.spawn(function()
            local source = ""
            local decompileOk = false

            pcall(function()
                source = target.Source
                if source and #source > 0 then
                    decompileOk = true
                end
            end)

            if not decompileOk then
                pcall(function()
                    source = decompile_fn(target)
                    if source and #source > 0 then
                        decompileOk = true
                    end
                end)
            end

            task.wait()
            self:_HideThinking()

            if not decompileOk or #source == 0 then
                self:_AddMessage("ai", "Could not decompile `" .. tostring(target.Name) .. "`.\n\nThis may be because:\n- The script is obfuscated\n- Your executor doesn't support decompilation\n- The script is protected")
                return
            end

            local displaySource = #source > 8000 and (source:sub(1, 8000) .. "\n-- [truncated, " .. #source .. " total chars]") or source
            self:_AddMessage("ai", "**Decompiled: `" .. tostring(target.Name) .. "` (" .. cls .. ")**\n```lua\n" .. displaySource .. "\n```")

            if self._client then
                local snippet = #source > 3000 and source:sub(1, 3000) .. "\n-- [truncated]" or source
                self._client:AddContext("user", "I decompiled " .. pathStr .. " (" .. cls .. "). Here is the source:\n```lua\n" .. snippet .. "\n```\nAnalyze it.")
                self._client:AddContext("model", "I can see the decompiled source. Ready to analyze it — what would you like to know?")
                self:_AddMessage("ai", "I've loaded the script source into context. Ask me to analyze, explain, or modify it!")
            end
        end)
    end)
end
